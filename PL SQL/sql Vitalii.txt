SELECT * FROM GERARD.FIELD_CONFIG FC
LEFT JOIN GERARD.FIELD_SYS_CONFIG FSC ON (FC.FIELDID = FSC.FIELDID)
LEFT JOIN GERARD.FIELD_CONFIG_GROUP FCG ON (FSC.SYSFDID = FCG.SYSFDID)
LEFT JOIN GERARD.UTIL_LABELS UL ON (FC.DISPLAYNAME = UL.KEY)
WHERE FC.FIELDID LIKE '%L' AND LOWER(UL.STD) like '%silent%';

SELECT MAX(FIELDID) FROM GERARD.FIELD_CONFIG
WHERE FIELDID LIKE '%L';

SELECT MAX(SYSFDID) FROM GERARD.FIELD_SYS_CONFIG
WHERE SYSFDID LIKE '%L';

SELECT SILENTCONFIRMATIONINDICATOR FROM GERARD.EXPORT_LOC_APPLICATION;
